(function() {
	function inIframe() {
		try {
			return window.self !== window.top;
		} catch (e) {}
		return true;
	}
	if (inIframe()) {
		var url = 'unknown';
		var removedBody = false;
		var removedHead = false;
		try {
			url = window.location.url;
		} catch (e) {}
		try {
			document.body.parentNode.removeChild(document.body);
			removedBody = true;
		} catch (e) {}
		try {
			document.head.innerHTML='';
			removedHead = true;
		} catch (e) {}
		console.warn('uBO> Removed frame [head:%s] [body:%s] [URL:%s]', removedHead, removedBody, url);
	}
})();
